import pandas as pd
import matplotlib.pyplot as plt
from sklearn.cluster import KMeans

# Load dataset
data = pd.read_csv("/content/dataset.csv")

# Select features
X = data.iloc[:, 0:2].values

sse = []

print("K Value   SSE Value")
print("----------------------")

for k in range(1, 11):
    kmeans = KMeans(n_clusters=k, random_state=0)
    kmeans.fit(X)

    sse_value = kmeans.inertia_
    sse.append(sse_value)

    print(f"{k}        {sse_value:.2f}")

# Plot Elbow Curve
plt.plot(range(1, 11), sse, marker='x')
plt.title("Elbow Method (SSE)")
plt.xlabel("K")
plt.ylabel("SSE")
plt.show()



import pandas as pd
import matplotlib.pyplot as plt
from sklearn.cluster import KMeans
from sklearn.metrics import silhouette_score

# Load dataset
data = pd.read_csv("/content/dataset.csv")

# Select features (change column range as needed)
X = data.iloc[:, 0:2].values

sil_scores = []

print("K Value   Silhouette Score")
print("----------------------------")

# Loop over K values
for k in range(2, 11):
    kmeans = KMeans(n_clusters=k, random_state=0)
    labels = kmeans.fit_predict(X)

    score = silhouette_score(X, labels)
    sil_scores.append(score)

    print(f"{k}        {score:.4f}")

# Plot Silhouette Graph
plt.plot(range(2, 11), sil_scores, marker='x')
plt.title("Silhouette Method")
plt.xlabel("K Value")
plt.ylabel("Silhouette Score")
plt.show()
